void ${class}::start() /* override final */
{
    std::cout << "BEGIN ${class}::start\n";
    assert(group);
    if (auto g = group->get())
    {
        int gx = g->x();
        int gy = g->y();
        int gw = g->w();
        int gh = g->h();
        if (g->as_window())
        {
            gx = 0;
            gy = 0;
        }
        int x = gx + 10;
        int y = gy + 10;
        int w = gw - 20;
        int h = 20;
        int lh = h + 5;
        auto innerGroup = widget<${class}, Fl_Group>(x, y += 10, w, gh - 30, "GenActionCode", nullptr);
        innerGroup->get()->box(Fl_Boxtype::FL_DOWN_BOX);
        x += 10;
        w -= 20;
        y += 10;

        cancelB = button(x, y, w, h, "Cancel", &${class}::onCancel);
        typedef void (${class}::*MyMethodPtr)();
        nameInput = widget<${class}, Fl_Input>(x + 100, y += lh, w - 100, h, "Klassenname", &${class}::onInput);
        nameInput->get()->when(FL_WHEN_ENTER_KEY | FL_WHEN_NOT_CHANGED);
        innerGroup->get()->end();
        std::cout << "end ${class}::start\n";
    }
}
